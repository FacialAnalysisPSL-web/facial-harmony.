<script>

const video = document.getElementById("video")
const canvas = document.getElementById("canvas")
const ctx = canvas.getContext("2d")

navigator.mediaDevices.getUserMedia({video:true})
.then(stream=>{
video.srcObject = stream
})

const faceMesh = new FaceMesh({
locateFile:(file)=>{
return `https://cdn.jsdelivr.net/npm/@mediapipe/face_mesh/${file}`
}
})

faceMesh.setOptions({
maxNumFaces:1,
refineLandmarks:true,
minDetectionConfidence:0.5,
minTrackingConfidence:0.5
})

faceMesh.onResults(onResults)

video.onloadeddata = startDetect

async function startDetect(){

await faceMesh.send({image:video})
requestAnimationFrame(startDetect)

}

function distance(a,b){

return Math.sqrt(
Math.pow(a.x-b.x,2)+
Math.pow(a.y-b.y,2)
)

}

function angle(a,b){

return Math.atan2(b.y-a.y,b.x-a.x)*(180/Math.PI)

}

function onResults(results){

ctx.clearRect(0,0,canvas.width,canvas.height)
ctx.drawImage(results.image,0,0,canvas.width,canvas.height)

if(results.multiFaceLandmarks.length>0){

const l = results.multiFaceLandmarks[0]

drawPoints(l)
drawLines(l)

let ratios = calculateRatios(l)

updateChart(ratios)

let score = harmonyScore(ratios)

document.getElementById("score").innerHTML =
"Harmony Score: "+score.toFixed(2)+"%"

}

}

function drawPoints(l){

for(let p of l){

ctx.beginPath()
ctx.arc(p.x*640,p.y*480,1,0,2*Math.PI)
ctx.fillStyle="lime"
ctx.fill()

}

}

function drawLines(l){

let leftCheek = l[234]
let rightCheek = l[454]
let chin = l[152]
let forehead = l[10]

ctx.strokeStyle="red"
ctx.lineWidth=2

ctx.beginPath()
ctx.moveTo(leftCheek.x*640,leftCheek.y*480)
ctx.lineTo(rightCheek.x*640,rightCheek.y*480)
ctx.stroke()

ctx.beginPath()
ctx.moveTo(forehead.x*640,forehead.y*480)
ctx.lineTo(chin.x*640,chin.y*480)
ctx.stroke()

}

function calculateRatios(l){

let leftCheek = l[234]
let rightCheek = l[454]

let forehead = l[10]
let chin = l[152]

let eyeOuter = l[33]
let eyeInner = l[133]

let eyeOuterR = l[263]
let eyeInnerR = l[362]

let faceWidth = distance(leftCheek,rightCheek)
let faceHeight = distance(forehead,chin)

let FWHR = faceWidth/faceHeight

let eyeDistance = distance(eyeInner,eyeInnerR)

let ESR = eyeDistance/faceWidth*100

let tilt = angle(eyeInner,eyeOuter)

return{
FWHR:FWHR,
ESR:ESR,
Tilt:tilt
}

}

let radarChart

function updateChart(r){

let data = [

r.FWHR,
r.ESR/50,
Math.abs(r.Tilt)/10,
1.1,
1.0

]

if(!radarChart){

radarChart = new Chart(document.getElementById("chart"),{

type:"radar",

data:{
labels:[
"FWHR",
"Eye Separation",
"Canthal Tilt",
"Jaw",
"Midface"
],

datasets:[{
label:"Harmony Ratios",
data:data
}]
}

})

}else{

radarChart.data.datasets[0].data=data
radarChart.update()

}

}

function harmonyScore(r){

let score = 0

if(r.FWHR>1.9 && r.FWHR<2.06) score+=20
else if(r.FWHR>1.8) score+=15
else score+=5

if(r.ESR>44 && r.ESR<47) score+=25
else score+=10

if(r.Tilt>5 && r.Tilt<9) score+=12
else score+=5

let maxScore = 333.71
let worstScore = -337.71

let normalized = (score-worstScore)/(maxScore-worstScore)*100

return normalized

}
